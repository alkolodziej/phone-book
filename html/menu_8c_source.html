369int strcasecmp_partial(const char *str1, const char *str2) {
370 // Loop through both strings until one of them reaches the end
371 while (*str1 && *str2) {
372 // Compare characters of both strings, ignoring case
373 if (tolower((unsigned char)*str1) != tolower((unsigned char)*str2))
374 return 0; // Return 0 if characters are not equal
375 str1++; // Move to the next character in str1
376 str2++; // Move to the next character in str2
377 }
378
379 // Check if str2 has been fully read
380 if (*str2 == '\0')
381 return 1; // Return true if str2 has been fully read, indicating str1 contains str2 as a substring
382 else
383 return 0; // Return false if str2 has not been fully read
384}
