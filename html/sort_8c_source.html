55void split(Contacts *source, Contacts **frontRef, Contacts **backRef) {
56 Contacts *fast;
57 Contacts *slow;
58
59 if (source == NULL || source->next == NULL) {
60 *frontRef = source;
61 *backRef = NULL;
62 } else {
63 slow = source;
64 fast = source->next;
65
66 // Advance 'fast' two nodes and advance 'slow' one node
67 while (fast != NULL) {
68 fast = fast->next;
69 if (fast != NULL) {
70 slow = slow->next;
71 fast = fast->next;
72 }
73 }
74

76 *frontRef = source;
77 *backRef = slow->next;
78 slow->next = NULL;
